33	class TreeSpecifierNode {
34		private TreeSpecifierNode parent = null;
35		private TreeSpecifierNode firstChild = null;
36		private TreeSpecifierNode nextSibling = null;
37		private Token tok;
38	
39	
40		TreeSpecifierNode(Token tok_)
41		{
42			tok = tok_;
43		}
44		public TreeSpecifierNode getFirstChild() { return firstChild; }
45		public TreeSpecifierNode getNextSibling() { return nextSibling; }
46		// Accessors
47		public TreeSpecifierNode getParent() { return parent; }
48		public Token getToken() { return tok; }
49		public void setFirstChild(TreeSpecifierNode child) {
50			firstChild = child;
51			child.parent = this;
52		}
53		// Structure-building
54		public void setNextSibling(TreeSpecifierNode sibling) {
55			nextSibling = sibling;
56			sibling.parent = parent;
57		}
58	}

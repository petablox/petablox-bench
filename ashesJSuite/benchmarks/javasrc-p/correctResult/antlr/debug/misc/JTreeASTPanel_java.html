1	package antlr.debug.misc;
2	
3	import java.awt.*;
4	import com.sun.java.swing.*;
5	import com.sun.java.swing.tree.*;
6	import com.sun.java.swing.event.*;
7	
8	public class JTreeASTPanel extends JPanel {
9	  JTree tree;
10	  
11	  public JTreeASTPanel(TreeModel tm, TreeSelectionListener listener) {
12		// use a layout that will stretch tree to panel size
13		setLayout(new BorderLayout());
14	
15		// Create tree
16		tree = new JTree(tm);
17	
18		// Change line style
19		tree.putClientProperty("JTree.lineStyle", "Angled");
20	
21		// Add TreeSelectionListener
22		if (listener != null)
23		  tree.addTreeSelectionListener (listener);
24	
25		// Put tree in a scrollable pane's viewport
26		JScrollPane sp = new JScrollPane();
27		sp.getViewport().add(tree);
28	
29		add(sp, BorderLayout.CENTER);
30	  }        
31	}

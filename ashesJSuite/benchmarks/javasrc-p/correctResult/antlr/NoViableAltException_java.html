34	import antlr.collections.AST;
35	
36	public class NoViableAltException extends ParserException {
37		public Token token;
38		public AST node;	// handles parsing and treeparsing
39	
40		public NoViableAltException(AST t) {
41			super("NoViableAlt");
42			node = t;
43		}
44		public NoViableAltException(Token t) {
45			super("NoViableAlt");
46			token = t;
47			line = t.getLine();
48		}
49		public String toString() {
50			if ( token!=null ) {
51				return "line("+token.getLine()+"), unexpected token: "+token.getText();
52			}
53			// must a tree parser error if token==null
54			if ( node==TreeParser.ASTNULL ) {
55				return "unexpected end of subtree";
56			}
57			return "unexpected AST node: "+node.toString();
58		}
59	}
